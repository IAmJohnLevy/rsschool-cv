                function HashStorageFunc () {
                  var self = this;
                  var drinks = {};
      
                  self.addValue = function addvalue(key, value) {
                      drinks[key] = value;
                  }
      
                  self.getValue = function getValue(key) {
                      if (key in drinks) {
                          return drinks[key];
                      } else return;
                  }
      
                  self.deleteValue = function deleteValue(key) {
                      if (key in drinks) {
                          delete drinks[key];
                          return true;
                      } else return false;
                  }
      
                  self.getKeys = function getKeys() {
                      return Object.keys(drinks);
                  } 
                }
